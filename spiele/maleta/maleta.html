// Theme direkt aus dem LocalStorage matchen if (localStorage.getItem('theme') === 'light') { document.documentElement.classList.add('light-mode'); } let alleVokabeln = []; let kofferReihenfolge = []; // Speichert die korrekte Abfolge der Vokabel-Objekte let aktuellerPruefSchritt = 0; // Wo befindet sich der Spieler beim Nachlegen? let kategorieName = "Kleidung"; // 1. Kategorie aus der URL holen const urlParams = new URLSearchParams(window.location.search); if (urlParams.has('kategorie')) { kategorieName = urlParams.get('kategorie'); } document.getElementById('current-category').innerText = kategorieName; // 2. Vokabeldaten laden fetch('../../vokabeln.json') .then(response => response.json()) .then(data => { const katData = data.find(k => k.kategorie === kategorieName); if (katData && katData.vokabeln.length >= 4) { alleVokabeln = katData.vokabeln; } else { document.getElementById('task-display').innerText = "Fehler!"; alert(`Kategorie "${kategorieName}" hat nicht genügend Vokabeln (mind. 4 benötigt).`); } }) .catch(err => { console.error(err); alert("vokabeln.json konnte nicht geladen werden."); }); function showOverlay(title, text, btnText) { document.getElementById('overlay-title').innerText = title; document.getElementById('overlay-text').innerText = text; document.getElementById('overlay-button').innerText = btnText; document.getElementById('game-overlay').style.display = 'flex'; } function closeOverlayAndStart() { document.getElementById('game-overlay').style.display = 'none'; if (alleVokabeln.length > 0 && kofferReihenfolge.length === 0) { neuesObjektHinzufuegen(); } else if (kofferReihenfolge.length > 0 && document.getElementById('overlay-button').innerText === "Erneut versuchen") { kofferReihenfolge = []; document.getElementById('kofferBox').innerHTML = 'Der Koffer ist noch leer...'; neuesObjektHinzufuegen(); } } // Fügt ein neues zufälliges Wort der Kette hinzu function neuesObjektHinzufuegen() { aktuellerPruefSchritt = 0; document.getElementById('round-display').innerText = kofferReihenfolge.length + 1; const randomVokabel = alleVokabeln[Math.floor(Math.random() * alleVokabeln.length)]; kofferReihenfolge.push(randomVokabel); // Phase 1: Dem Spieler das neue Wort zeigen document.getElementById('task-display').innerText = `Merke dir das neue Wort: "${randomVokabel.de}" (= ${randomVokabel.es})`; // Die Buttons kurz leeren, damit man nicht voreilig klickt document.getElementById('optionsGrid').innerHTML = ''; // Nach 2 Sekunden startet die Abfrage-Phase automatisch setTimeout(() => { starteAbfragePhase(); }, 2000); } function starteAbfragePhase() { // Phase 2: Text anpassen, je nachdem ob von vorn begonnen wird if (kofferReihenfolge.length === 1) { document.getElementById('task-display').innerText = `Packe das erste Wort in den Koffer!`; } else { document.getElementById('task-display').innerText = `Wiederhole die Kette! Was war das 1. Wort?`; } renderOptionButtons(); } function renderOptionButtons() { const grid = document.getElementById('optionsGrid'); grid.innerHTML = ''; // Bestimme das aktuell gesuchte Wort für diesen Klickschritt const gesuchtesWort = kofferReihenfolge[aktuellerPruefSchritt]; // Erstelle die Antwortmöglichkeiten (1 richtige, 3 falsche) let auswahl = [gesuchtesWort]; while (auswahl.length < Math.min(4, alleVokabeln.length)) { const falschesWort = alleVokabeln[Math.floor(Math.random() * alleVokabeln.length)]; if (!auswahl.some(item => item.id === falschesWort.id)) { auswahl.push(falschesWort); } } // Shuffle Auswahl-Array auswahl.sort(() => Math.random() - 0.5); auswahl.forEach(vokabel => { const btn = document.createElement('button'); btn.className = 'option-btn'; btn.textContent = vokabel.es; // Spanisch auf den Buttons btn.onclick = () => verifiziereAuswahl(vokabel); grid.appendChild(btn); }); } function verifiziereAuswahl(gewaehlteVokabel) { const korrekteVokabel = kofferReihenfolge[aktuellerPruefSchritt]; if (gewaehlteVokabel.id === korrekteVokabel.id) { aktuellerPruefSchritt++; // Wenn die gesamte Kette für diese Runde fehlerfrei durchgeklickt wurde if (aktuellerPruefSchritt === kofferReihenfolge.length) { document.getElementById('task-display').innerText = "Richtig! Der Koffer füllt sich..."; document.getElementById('optionsGrid').innerHTML = ''; aktualisiereKofferAnzeige(); setTimeout(() => { neuesObjektHinzufuegen(); }, 1200); } else { // Nächsten Gegenstand der Kette abfragen document.getElementById('task-display').innerText = `Richtig! Was kam nach "${korrekteVokabel.es}"?`; renderOptionButtons(); } } else { // Falsch getippt -> Game Over! showOverlay( "Game Over! 🧳💀", `Das war leider falsch. Richtig gewesen wäre "${korrekteVokabel.es}" (${korrekteVokabel.de}). Du hast eine Kette von ${kofferReihenfolge.length - 1} Begriffen geschafft!`, "Erneut versuchen" ); } } function aktualisiereKofferAnzeige() { const box = document.getElementById('kofferBox'); box.innerHTML = ''; kofferReihenfolge.forEach(item => { const span = document.createElement('span'); span.className = 'koffer-item'; span.textContent = item.es; box.appendChild(span); }); }
